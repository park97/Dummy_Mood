  <h1>How are you feeling today?</h1>

  <input type="text" id="moodInput" placeholder="e.g., happy, tired, excited">

  <br>

  <button onclick="showMood()">Submit</button>


  <p id="message"></p>


  <script>

    function showMood() {

      const mood = document.getElementById("moodInput").value.toLowerCase();

      const message = document.getElementById("message");


      if (mood === "happy") {

        message.textContent = "Yay! Keep shining 🌞";

      } else if (mood === "tired") {

        message.textContent = "Take a rest. You deserve it 💆‍♂️";

      } else if (mood === "excited") {

        message.textContent = "Go seize the day! 🚀";

      } else {

        message.textContent = "Thank you for sharing. Wishing you a peaceful day 💖";

      }

    }

  </script>
